<script>
$(document).ready(function () {
    // Simulasi load dokumen berdasarkan jenis (asumsi data tersedia via JSON)
    @php
        $suratList = isset($suratList) ? $suratList : [];
        $notaList = isset($notaList) ? $notaList : [];
    @endphp

    const suratList = @json($suratList);
    const notaList = @json($notaList);

    $('#jenis_dokumen').change(function () {
        const jenis = $(this).val();
        const select = $('#select_dokumen');
        select.empty().append('<option value="">Pilih Dokumen</option>');
        $('#hidden_jenis').val(jenis);

        if (jenis === 'surat') {
            suratList.forEach(item => {
                select.append(`<option value="${item.id}">${item.nomor_surat}</option>`);
            });
        } else if (jenis === 'nota') {
            notaList.forEach(item => {
                select.append(`<option value="${item.id}">${item.nomor_nota}</option>`);
            });
        }
    });

    // Isi modal edit saat tombol diklik
    $('.btn-edit').click(function () {
        // Di implementasi nyata, ini diisi via AJAX
        // Untuk refactor view, cukup siapkan struktur
    });
});
</script>
